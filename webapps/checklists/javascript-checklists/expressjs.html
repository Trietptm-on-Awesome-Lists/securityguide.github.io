    secure cookie options

    const express = require('express')
const session = require('express-session')

const app = express()
const hour = 3600000
app.use(session({
  cookie: { secure: true, sameSite: true, maxAge: hour }
}))

    secure header options

    const express = require('express')
const helmet = require('helmet')

const app = express()
app.use(helmet())
